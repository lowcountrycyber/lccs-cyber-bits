(function(){
  const input = document.getElementById('search-box');
  if (!input) return;
  const items = Array.from(document.querySelectorAll('li.searchable'));
  const counter = document.getElementById('search-results-count');
  function render() {
    const q = input.value.trim().toLowerCase();
    let visible = 0;
    items.forEach(li => {
      const text = (li.getAttribute('data-text') || '');
      const match = !q || text.includes(q);
      li.style.display = match ? '' : 'none';
      if (match) visible += 1;
    });
    if (counter) counter.textContent = q ? `${visible} matches` : '';
  }
  input.addEventListener('input', render);
})();
</script>
